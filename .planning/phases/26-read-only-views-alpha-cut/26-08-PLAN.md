use rand::{SeedableRng, Rng};
use rand::rngs::StdRng;
use std::fs;
use std::path::PathBuf;

#[test]
fn setup_perf_fixture() -> anyhow::Result<()> {
    let Ok(out) = std::env::var("PERF_FIXTURE_OUT") else {
        eprintln!("setup_perf_fixture: PERF_FIXTURE_OUT not set — skipping");
        return Ok(());
    };
    let out = PathBuf::from(out);
    let library = out.join("library");
    fs::create_dir_all(&library)?;

    let mut rng = StdRng::seed_from_u64(0xtome_perf);
    let mut manifest_entries = Vec::with_capacity(2000);

    for i in 0..2000 {
        let name = format!("skill-{i:04}");
        let dir = library.join(&name);
        fs::create_dir_all(&dir)?;
        let body_len: usize = rng.gen_range(100..5000);
        let body = generate_lorem(&mut rng, body_len);
        let content = format!("---\nname: {name}\ndescription: Synthetic test skill\n---\n\n{body}\n");
        fs::write(dir.join("SKILL.md"), &content)?;
        let content_hash = compute_sha256(&content);  // import sha2::Sha256, Digest
        manifest_entries.push(serde_json::json!({
            "name": name,
            "source_name": "synthetic",
            "content_hash": content_hash,
            "synced_at": "2026-05-29T00:00:00Z",
        }));
    }

    let manifest = serde_json::json!({ "entries": manifest_entries });
    fs::write(out.join(".tome-manifest.json"), serde_json::to_string_pretty(&manifest)?)?;

    let tome_toml = format!(r#"[directories.synthetic]
type = "directory"
role = "managed"
path = "{}/library"
"#, out.display());
    fs::write(out.join("tome.toml"), tome_toml)?;

    eprintln!("perf fixture written to {} (2000 skills)", out.display());
    Ok(())
}

fn generate_lorem(rng: &mut impl Rng, target_len: usize) -> String { /* repeat lorem ipsum to ~target_len */ }
fn compute_sha256(content: &str) -> String { /* hex sha256 */ }
```

declare global {
  interface Window {
    __fpsFrames: number[];
  }
}

window.__fpsFrames = [];
const start = performance.now();
let last = start;
function tick(t: number) {
  window.__fpsFrames.push(t - last);
  last = t;
  if (t - start < 2000) requestAnimationFrame(tick);
}

import { test, expect } from '@playwright/test';
import fs from 'fs';
import path from 'path';

test('2000 skills search-as-you-type sustains 60fps p95', async ({ page }) => {
  await page.addInitScript({ path: path.join(__dirname, 'fps-sampler.ts') });
  await page.goto('http://localhost:5173/?view=skills');
  await page.waitForSelector('[role="option"]', { timeout: 10000 });
  await page.focus('[aria-label="Search skills"]');
  await page.keyboard.type('tdd', { delay: 15 });
  await page.waitForTimeout(2200);  // 2s sampling + headroom

  const frames: number[] = await page.evaluate(() => window.__fpsFrames);
  expect(frames.length).toBeGreaterThan(30);  // need real samples

  const sorted = [...frames].sort((a, b) => a - b);
  const p95 = sorted[Math.floor(sorted.length * 0.95)];
  console.log(`p95 inter-frame: ${p95.toFixed(2)}ms (target: < 18ms ≈ 55fps)`);

  // Write baseline to 26-PERF-REPORT.md for the human to inspect
  const median = sorted[Math.floor(sorted.length * 0.5)];
  const max = sorted[sorted.length - 1];
  const reportLine = `\n- run ${new Date().toISOString()}: p50=${median.toFixed(2)}ms p95=${p95.toFixed(2)}ms max=${max.toFixed(2)}ms (samples=${frames.length})`;
  fs.appendFileSync(path.join(__dirname, '../../../.planning/phases/26-read-only-views-alpha-cut/26-PERF-REPORT.md'), reportLine);

  expect(p95).toBeLessThan(18);
});
```
