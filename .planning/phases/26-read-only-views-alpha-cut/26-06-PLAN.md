```rust
//! File watcher — manifest / lockfile / library / machine.toml.
//!
//! Rust-side `notify 8.2` + `notify-debouncer-full 0.7` watching the four
//! roots that share state with the CLI (NF-05). Each filesystem change
//! emits one of four typed `tauri-specta` events the React side subscribes
//! to via `useTauriEvent` (plan 26-06). Debounce window 200ms (Pitfall 1
//! — atomic-rename read race; matches SC#1's 200ms refresh target).
//!
//! OQ-3 resolution: Rust-side rather than JS `@tauri-apps/plugin-fs::watch`
//! — typed events route React refetches correctly; the IPC surface stays
//! auditable (no `fs:default` permission widening).

use anyhow::{Context, Result};
use notify::RecursiveMode;
use notify_debouncer_full::{new_debouncer, DebouncedEvent};
use std::time::Duration;
use tauri_specta::Event;

#[derive(Clone, Debug, serde::Serialize, specta::Type, tauri_specta::Event)]
pub struct ManifestChanged;
#[derive(Clone, Debug, serde::Serialize, specta::Type, tauri_specta::Event)]
pub struct LockfileChanged;
#[derive(Clone, Debug, serde::Serialize, specta::Type, tauri_specta::Event)]
pub struct LibraryChanged;
#[derive(Clone, Debug, serde::Serialize, specta::Type, tauri_specta::Event)]
pub struct MachinePrefsChanged;

pub fn spawn_watcher(app: tauri::AppHandle, paths: tome::TomePaths) -> Result<()> {
    let manifest_path = paths.config_dir().join(".tome-manifest.json");
    let lockfile_path = paths.config_dir().join("tome.lock");
    let library_dir = paths.library_dir().to_path_buf();
    let machine_path = tome::machine::default_machine_path()
        .context("failed to resolve default machine.toml path")?;

    let app2 = app.clone();
    std::thread::spawn(move || {
        if let Err(e) = run_watcher(app2, manifest_path, lockfile_path, library_dir, machine_path) {
            eprintln!("watcher thread exited: {e:#}");
        }
    });
    Ok(())
}

fn run_watcher(
    app: tauri::AppHandle,
    manifest_path: std::path::PathBuf,
    lockfile_path: std::path::PathBuf,
    library_dir: std::path::PathBuf,
    machine_path: std::path::PathBuf,
) -> Result<()> {
    let (tx, rx) = std::sync::mpsc::channel();
    let mut debouncer = new_debouncer(
        Duration::from_millis(200),
        None,
        move |res: Result<Vec<DebouncedEvent>, _>| {
            if let Ok(events) = res {
                let _ = tx.send(events);
            }
        },
    ).context("failed to init notify debouncer")?;

    // Watch PARENT dirs, not file paths — Pitfall 5 (notify::watch fails on
    // non-existent paths; parent dirs exist or are creatable).
    let watch_targets: Vec<(std::path::PathBuf, RecursiveMode)> = vec![
        (manifest_path.parent().map(|p| p.to_path_buf()).unwrap_or_else(|| std::path::PathBuf::from(".")), RecursiveMode::NonRecursive),
        (lockfile_path.parent().map(|p| p.to_path_buf()).unwrap_or_else(|| std::path::PathBuf::from(".")), RecursiveMode::NonRecursive),
        (library_dir.clone(), RecursiveMode::Recursive),
        (machine_path.parent().map(|p| p.to_path_buf()).unwrap_or_else(|| std::path::PathBuf::from(".")), RecursiveMode::NonRecursive),
    ];
    for (path, mode) in &watch_targets {
        if path.exists() {
            if let Err(e) = debouncer.watch(path, *mode) {
                eprintln!("warn: failed to watch {}: {e:#}", path.display());
            }
        }
    }

    while let Ok(events) = rx.recv() {
        let mut saw_manifest = false;
        let mut saw_lockfile = false;
        let mut saw_library = false;
        let mut saw_machine = false;
        for ev in events {
            for path in &ev.paths {
                if path == &manifest_path { saw_manifest = true; }
                if path == &lockfile_path { saw_lockfile = true; }
                if path == &machine_path { saw_machine = true; }
                if path.starts_with(&library_dir) { saw_library = true; }
            }
        }
        if saw_manifest { let _ = ManifestChanged.emit(&app); }
        if saw_lockfile { let _ = LockfileChanged.emit(&app); }
        if saw_library { let _ = LibraryChanged.emit(&app); }
        if saw_machine { let _ = MachinePrefsChanged.emit(&app); }
    }

    Ok(())
}
```

```tsx
import { useEffect } from 'react';

type EventListener = { listen: (cb: () => void) => Promise<() => void> };

export function useTauriEvent(event: EventListener, handler: () => void) {
  useEffect(() => {
    let unlisten: (() => void) | null = null;
    let cancelled = false;
    event.listen(() => { if (!cancelled) handler(); }).then((un) => {

}
```

```tsx
import { useTauriEvent } from './useTauriEvent';

export function useStatus() {
  const [status, setStatus] = useState<StatusReport | null>(null);
  const [err, setErr] = useState<TomeError | null>(null);

  useTauriEvent(events.lockfileChanged, refetch);
  useTauriEvent(events.libraryChanged, refetch);
  useTauriEvent(events.machinePrefsChanged, refetch);

  return { status, err, updatedAt, refetch };
}
```
